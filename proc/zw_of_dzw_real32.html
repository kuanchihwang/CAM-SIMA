    pure function zw_of_dzw_real32(dzw) result(zw)
        use, intrinsic :: iso_fortran_env, only: real32

        real(real32), intent(in) :: dzw(:)
        real(real32) :: zw(size(dzw) + 1)

        integer :: k

        do k = 1, size(zw)
            zw(k) = sum(dzw(1:k - 1))
        end do
    end function zw_of_dzw_real32
