    pure elemental function t_by_equation_of_state(constant_r, p, rho) result(t)
        use, intrinsic :: iso_fortran_env, only: real64

        real(real64), intent(in) :: constant_r, p, rho
        real(real64) :: t

        t = p / (rho * constant_r)
    end function t_by_equation_of_state
